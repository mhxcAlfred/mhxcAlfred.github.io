# ------------------------------------------
# 针对百度蜘蛛 (Baiduspider) 的特殊规则
# ------------------------------------------
User-agent: Baiduspider					# 指定以下规则仅对百度的网络爬虫生效
Disallow: /						# 完全禁止百度蜘蛛抓取本站的任何页面和资源（全站屏蔽）

# ------------------------------------------
# 针对所有其他搜索引擎的通用规则
# ------------------------------------------
User-agent: *						# 指定以下规则对除百度外的所有搜索引擎爬虫生效（如 Google, Bing, DuckDuckGo 等）

# --- 禁止抓取的目录与文件（剔除无 SEO 价值或涉及隐私的资源） ---
Disallow: /vendors/					# 禁止爬虫抓取第三方依赖库目录，避免浪费网站的抓取配额
Disallow: /fancybox/					# 禁止爬虫抓取图片灯箱插件目录，该目录无网页文本内容
Disallow: /fonts/						# 禁止爬虫抓取字体文件目录，搜索引擎不需要索引字体
Disallow: /image/						# 禁止爬虫抓取静态图片资源目录，防止博客图片被单独索引到图片搜索中（可按需调整）
Disallow: /downloads/					# 禁止爬虫抓取下载文件目录，保护文件并节省服务器带宽
Disallow: /tmp/						# 禁止爬虫抓取临时文件目录，防止未完成的草稿或缓存被公开
Disallow: /search.xml					# 禁止爬虫抓取本地搜索生成的 XML 数据文件，防止全站内容在搜索结果中裸露
Disallow: /search.json					# 禁止爬虫抓取本地搜索生成的 JSON 数据文件，防止全站内容在搜索结果中裸露

# --- 显式允许抓取的关键路径（确保核心页面被正常索引） ---
Allow: /							# 默认允许抓取除上方 Disallow 之外的所有根目录及子页面
Allow: /404/						# 允许抓取 404 错误提示页面
Allow: /about/						# 允许抓取“关于我”页面
Allow: /anniversary/					# 允许抓取“纪念日”分类或专题页面
Allow: /archives/						# 允许抓取博客文章归档列表页
Allow: /categories/						# 允许抓取博客文章分类列表页
Allow: /tags/						# 允许抓取博客文章标签列表页
Allow: /page/						# 允许抓取博客的分页列表（如主页的第2页、第3页等）
Allow: /zh-CN/						# 允许抓取中文语系专属的页面或子目录
